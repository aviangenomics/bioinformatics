
#Visualizes interactive quality plot to determine where to trim reads. The trim locations should exclude low quality reads and keep high quality reads.
qiime tools view demux.qzv

#Trims at the specified values to get rid of low quality reads.
qiime dada2 denoise-paired \
  --i-demultiplexed-seqs demux.qza \
  --p-trunc-len-f 190 \
  --p-trunc-len-r 190 \
  --o-representative-sequences rep-seqs-dada2.qza \
  --o-table table-dada2.qza \
  --o-denoising-stats stats-dada2.qza

#Converts metadata into a QIIME visualization file.
qiime metadata tabulate \
  --m-input-file stats-dada2.qza \
  --o-visualization stats-dada2.qzv

#Renames files.
mv rep-seqs-dada2.qza rep-seqs.qza
mv table-dada2.qza table.qza

#Converts .qza files to .qzv files that can be visualized using qiime tools view. The table.qzv file can be used to determine sequencing depth.
qiime feature-table summarize \
  --i-table table.qza \
  --o-visualization table.qzv \
  --m-sample-metadata-file sample-metadata.tsv
qiime feature-table tabulate-seqs \
  --i-data rep-seqs.qza \
  --o-visualization rep-seqs.qzv

#Aligns sequences to produce an unrooted phylogeny to be used for assigning taxonomy.
qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences rep-seqs.qza \
  --o-alignment aligned-rep-seqs.qza \
  --o-masked-alignment masked-aligned-rep-seqs.qza \
  --o-tree unrooted-tree.qza \
  --o-rooted-tree rooted-tree.qza

#Creates Shannon Diversity visualization for genets, body-site, and symbionts.
qiime diversity alpha-group-significance \
  --i-alpha-diversity core-metrics-results/shannon_vector.qza \
  --m-metadata-file sample-metadata.tsv \
  --o-visualization core-metrics-results/shannon-group-significance.qzv

#Creates Observed Features visualization for all samples.
qiime diversity alpha-group-significance \
  --i-alpha-diversity core-metrics-results/observed_features_vector.qza \
  --m-metadata-file sample-metadata.tsv \ 
  --o-visualization core-metrics-results/observed-features-group-significance.qzv

#Creates Beta Diversity visualization for genet
qiime diversity beta-group-significance \
  --i-distance-matrix core-metrics-results/bray_curtis_distance_matrix.qza \
  --m-metadata-file sample-metadata.tsv \
  --m-metadata-column genet \
  --o-visualization core-metrics-results/bray-curtis-genet-significance.qzv \
  --p-pairwise

#Creates Beta Diversity visualization for body-site
qiime diversity beta-group-significance \
  --i-distance-matrix core-metrics-results/bray_curtis_distance_matrix.qza \
  --m-metadata-file sample-metadata.tsv \
  --m-metadata-column body-site \
  --o-visualization core-metrics-results/bray-curtis-bsb-significance.qzv \
  --p-pairwise

#Creates Beta Diversity visualization for symbiont
qiime diversity beta-group-significance \
  --i-distance-matrix core-metrics-results/bray_curtis_distance_matrix.qza \
  --m-metadata-file sample-metadata.tsv \
  --m-metadata-column symbiont \
  --o-visualization core-metrics-results/bray-curtis-symbiont-significance.qzv \
  --p-pairwise

#Assign taxonomy using gg-13-8-99-515-806-nb-classifier.qza
qiime feature-classifier classify-sklearn \
  --i-classifier gg-13-8-99-515-806-nb-classifier.qza \
  --i-reads rep-seqs.qza \
  --o-classification taxonomy.qza

qiime metadata tabulate \
  --m-input-file taxonomy.qza \
  --o-visualization taxonomy.qzv

#Create interactive barplot to examine taxonomic groups present
qiime taxa barplot \
  --i-table table.qza \
  --i-taxonomy taxonomy.qza \
  --m-metadata-file metadata.tsv \
  --o-visualization taxa-bar-plots.qzv
